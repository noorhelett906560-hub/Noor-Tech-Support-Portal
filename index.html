// 1. Database Setup (Browser ki memory) let crmStorage = JSON.parse(localStorage.getItem('noor_ultimate_db')) || []; // 2. Dashboard Update Karne ka Function function updateUI() { document.getElementById('countTotal').innerText = crmStorage.length; document.getElementById('countPending').innerText = crmStorage.filter(t => t.status === 'Pending').length; document.getElementById('countReplace').innerText = crmStorage.filter(t => t.type === 'Replacement').length; document.getElementById('countResolved').innerText = crmStorage.filter(t => t.status === 'Resolved').length; const tableBody = document.getElementById('crmTableBody'); tableBody.innerHTML = ''; crmStorage.forEach((item) => { tableBody.innerHTML += ` #${item.id} ${item.name}
${item.phone} ${item.sku}
${item.type} Inv: ${item.inv}
SN: ${item.sn} ${item.status} ${item.priority} ${item.date} `; }); } // 3. Form Submit Karne ka Function (Yahan Data Save Hota Hai) document.getElementById('crmForm').addEventListener('submit', function(e) { e.preventDefault(); // Data Object Banana const ticket = { id: Math.floor(1000 + Math.random() * 9000), name: document.getElementById('custName').value, phone: document.getElementById('custPhone').value, inv: document.getElementById('invNo').value, sku: document.getElementById('sku').value, sn: document.getElementById('serial').value, type: document.getElementById('issueType').value, priority: document.getElementById('priority').value, warranty: document.getElementById('warranty').value, status: document.getElementById('status').value, date: new Date().toLocaleDateString(), note: document.getElementById('issueNote').value }; // BROWSER MEIN SAVE KARO crmStorage.unshift(ticket); localStorage.setItem('noor_ultimate_db', JSON.stringify(crmStorage)); // GOOGLE SHEET MEIN BHEJO (Apna URL niche paste karein) fetch('APNA_SCRIPT_URL_YAHAN_PASTE_KAREIN', { method: 'POST', mode: 'no-cors', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body: new URLSearchParams(ticket) }); alert("Mubarak ho Noor bhai! Ticket Save ho gaya."); updateUI(); this.reset(); }); // Pehli baar load hone par data dikhao updateUI();
